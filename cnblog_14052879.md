char l[n]; main() { //处理一行 for (i = 0; (c = getchar()) != EOF && c != '\n';
++i) { l[i] = c; printf("i=%d\t1=%c\n", i, l[i]); } printf("%d\n", i); if (c
== '\n') { l[i] = c; printf("i=%d\t2=%c\n", i, l[i]); i++; } l[i] = '\0';
printf("i=%d\t3=%c", i, l[i]); return i; } ``` 随便输入
